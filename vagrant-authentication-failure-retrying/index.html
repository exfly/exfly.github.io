# term1:
vagrant up # 执行完后切换到 term2

# term2:
vagrant ssh
echo 'PubkeyAcceptedKeyTypes=+ssh-rsa' >> /etc/ssh/sshd_config
systemctl restart sshd

# term1
# 可以看到日志, 可以看到问题已经解决了
